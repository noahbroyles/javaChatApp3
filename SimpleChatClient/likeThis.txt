/***
* fills up an arraylist by reading LINES from the keyboard
* @param a the arraylist to  fill up
* @return the filled arraylist
*/
public static ArrayList<String> fillArrayList(ArrayList<String> a) {
	// Keep it from howling in yellow
	@SuppressWarnings("resource")
	Scanner readKBD = new Scanner(System.in);
	System.out.println("How many names do you want to add to the ArrayList? ");
	int numberOfNamesLeftToAdd = readKBD.nextInt();
	int added = 0;
	while (added < numberOfNamesLeftToAdd) {
		System.out.println("Enter name #" + (added + 1) +": ");
		a.add(readKBD.nextLine());
		added++;
	}

	return a;
}
